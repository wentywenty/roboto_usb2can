57static void can_state_change_callback(const struct device *dev, enum can_state state,
58 struct can_bus_err_cnt err_cnt, void *user_data)
59{
60 int ch = 0; /* Currently only supports single channel, multi-channel needs matching via
61 user_data or dev */
62 struct can_error_monitor *mon = &err_monitors[ch];
63 int64_t now = k_uptime_get();
64
65 /* Reset statistics window */
66 if ((now - mon->window_start_ms) > CAN_ERR_WINDOW_MS) {
67 mon->err_frame_count = 0;
68 mon->window_start_ms = now;
69 mon->throttled = false;
70 }
71
72 /* Error frame count */
73 if (state != CAN_STATE_ERROR_ACTIVE) {
74 mon->err_frame_count++;
75 }
76
77 /* Error frame flood detection */
78 if (mon->err_frame_count > CAN_ERR_FRAME_THRESHOLD) {
79 if (!mon->throttled) {
80 LOG_ERR("CH%d: Error frame flood detected (%u/s), throttling", ch,
81 mon->err_frame_count);
82 mon->throttled = true;
83 /* USB error and CAN error */
84 status_led_usb_set(LED_USB_ERROR);
85 status_led_can_set(CAN_LED_ERROR);
86 }
87 /* Actively enter Bus-Off to protect bus */
88 if (!mon->forced_busoff) {
89 LOG_ERR("CH%d: Forcing Bus-Off to prevent bus freeze", ch);
90 can_stop(dev);
91 mon->forced_busoff = true;
92 return;
93 }
94 }
95
96 /* Handle different error states */
97 switch (state) {
98 case CAN_STATE_ERROR_ACTIVE:
99 LOG_DBG("CH%d: CAN ERROR_ACTIVE (TEC=%u, REC=%u)", ch, err_cnt.tx_err_cnt,
100 err_cnt.rx_err_cnt);
101 mon->err_passive_count = 0; /* Reset count after recovery */
102 mon->forced_busoff = false;
103 /* CAN back to normal */
104 status_led_can_set(CAN_LED_ACTIVE);
105 break;
106
107 case CAN_STATE_ERROR_WARNING:
108 LOG_WRN("CH%d: CAN ERROR_WARNING (TEC=%u, REC=%u)", ch, err_cnt.tx_err_cnt,
109 err_cnt.rx_err_cnt);
110 /* CAN warning state */
111 status_led_can_set(CAN_LED_WARNING);
112 break;
113
114 case CAN_STATE_ERROR_PASSIVE:
115 mon->err_passive_count++;
116 LOG_WRN("CH%d: CAN ERROR_PASSIVE #%u (TEC=%u, REC=%u)", ch, mon->err_passive_count,
117 err_cnt.tx_err_cnt, err_cnt.rx_err_cnt);
118 /* CAN error state */
119 status_led_can_set(CAN_LED_ERROR);
120
121 /* Persistent ERROR_PASSIVE -> active Bus-Off */
122 if (mon->err_passive_count > CAN_ERR_PASSIVE_LIMIT) {
123 LOG_ERR("CH%d: Persistent ERROR_PASSIVE (%u times), forcing Bus-Off", ch,
124 mon->err_passive_count);
125 can_stop(dev);
126 mon->forced_busoff = true;
127 /* USB error and CAN off */
128 status_led_usb_set(LED_USB_ERROR);
129 status_led_can_set(CAN_LED_OFF);
130 }
131 break;
132
133 case CAN_STATE_BUS_OFF:
134 LOG_ERR("CH%d: CAN BUS_OFF (TEC=%u, REC=%u)", ch, err_cnt.tx_err_cnt,
135 err_cnt.rx_err_cnt);
136 mon->forced_busoff = true;
137 /* USB error and CAN off */
138 status_led_usb_set(LED_USB_ERROR);
139 status_led_can_set(CAN_LED_OFF);
140 break;
141
142 case CAN_STATE_STOPPED:
143 LOG_INF("CH%d: CAN STOPPED", ch);
144 /* CAN stopped */
145 status_led_can_set(CAN_LED_OFF);
146 break;
147 }
148}
