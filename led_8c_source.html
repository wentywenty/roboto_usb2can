33/* USB LED blink work (Blue LED) */
34static void usb_led_blink_work(struct k_work *work)
35{
36 const struct led_pattern *pattern = &usb_led_patterns[current_usb_status];
37 uint32_t next_delay;
38
39 /* Toggle LED state */
40 usb_led_state = !usb_led_state;
41 gpio_pin_set_dt(&usb_led, usb_led_state);
42
43 /* Calculate next trigger time */
44 if (usb_led_state) {
45 next_delay = pattern->on_ms;
46 } else {
47 next_delay = pattern->off_ms;
48
49 /* Check repeat count */
50 if (pattern->repeat > 0) {
51 usb_repeat_count++;
52 if (usb_repeat_count >= pattern->repeat) {
53 /* Completed specified count, stop */
54 gpio_pin_set_dt(&usb_led, 0);
55 return;
56 }
57 }
58 }
59
60 /* Reschedule work */
61 k_work_reschedule(&usb_led_work, K_MSEC(next_delay));
62}

101/* Activity LED timer callback (triggers every 50ms) */
102static void activity_tick_handler(struct k_timer *timer)
103{
104 ARG_UNUSED(timer);
105
106 if (activity_ticks > 0) {
107 activity_ticks--;
108 if (activity_ticks == (LED_TICKS_ACTIVITY / 2)) {
109 /* Turn on after middle tick */
110 gpio_pin_set_dt(&activity_led, 1);
111 } else if (activity_ticks == 0) {
112 /* Turn off at end */
113 gpio_pin_set_dt(&activity_led, 0);
114 }
115 }

118/* Initialize status LED */
119int status_led_init(void)
120{
121 int ret;
122
123 /* Initialize USB LED (Blue) */
124 if (!gpio_is_ready_dt(&usb_led)) {
125 LOG_ERR("USB LED GPIO not ready");
126 return -ENODEV;
127 }
128
129 ret = gpio_pin_configure_dt(&usb_led, GPIO_OUTPUT_INACTIVE);
130 if (ret < 0) {
131 LOG_ERR("Failed to configure USB LED: %d", ret);
132 return ret;
133 }
134
135 /* Initialize Activity LED (Green) */
136 if (!gpio_is_ready_dt(&activity_led)) {
137 LOG_WRN("Activity LED GPIO not ready");
138 } else {
139 ret = gpio_pin_configure_dt(&activity_led, GPIO_OUTPUT_INACTIVE);
140 if (ret < 0) {
141 LOG_WRN("Failed to configure activity LED: %d", ret);
142 } else {
143 k_work_init_delayable(&activity_work, activity_led_off_work);
144 LOG_INF("Activity LED initialized");
145 }
146 }
147
148 /* Initialize CAN LED (Yellow) */
149 if (!gpio_is_ready_dt(&can_led)) {
150 LOG_WRN("CAN LED GPIO not ready");
151 } else {
152 ret = gpio_pin_configure_dt(&can_led, GPIO_OUTPUT_INACTIVE);
153 if (ret < 0) {
154 LOG_WRN("Failed to configure CAN LED: %d", ret);
155 } else {
156 k_work_init_delayable(&can_led_work, can_led_blink_work);
157 LOG_INF("CAN LED initialized");
158 }
159 }
160
161 /* Initialize work queues */
162 k_work_init_delayable(&usb_led_work, usb_led_blink_work);
163
164 /* Initialize activity LED timer */
165 k_timer_init(&activity_tick_timer, activity_tick_handler, NULL);
166 k_timer_start(&activity_tick_timer, K_MSEC(LED_TICK_MS), K_MSEC(LED_TICK_MS));
167 last_activity_time = sys_timepoint_calc(K_NO_WAIT);
168 last_stopped_time = sys_timepoint_calc(K_NO_WAIT); /* Initialize STOPPED filter */
169
170 /* Start with USB ready state */
171 status_led_usb_set(LED_USB_READY);
172 status_led_can_set(CAN_LED_OFF);
173
174 LOG_INF("Status LEDs initialized");
175 return 0;
176}

178/* Set USB LED status (Blue LED) */
179void status_led_usb_set(enum led_status status)
180{
181 if (status >= ARRAY_SIZE(usb_led_patterns)) {
182 LOG_ERR("Invalid USB LED status: %d", status);
183 return;
184 }
185
186 /* Cancel current blink */
187 k_work_cancel_delayable(&usb_led_work);
188
189 current_usb_status = status;
190 usb_repeat_count = 0;
191 usb_led_state = false;
192
193 /* Start new blink pattern */
194 const struct led_pattern *pattern = &usb_led_patterns[status];
195 gpio_pin_set_dt(&usb_led, 1);
196 k_work_reschedule(&usb_led_work, K_MSEC(pattern->on_ms));
197
198 LOG_DBG("USB LED status changed to %d", status);
199}

235/* gs_usb event callback function */
236int status_led_event(const struct device *dev, uint16_t ch, enum gs_usb_event event,
237 void *user_data)
238{
239 ARG_UNUSED(dev);
240 ARG_UNUSED(ch);
241 ARG_UNUSED(user_data);
242
243 switch (event) {
244 case GS_USB_EVENT_CHANNEL_ACTIVITY_RX:
245 __fallthrough;
246 case GS_USB_EVENT_CHANNEL_ACTIVITY_TX:
247 /* Low-pass filter: prevent LED from blinking too fast */
248 if (!sys_timepoint_expired(last_activity_time)) {
249 return 0; /* Ignore frequent events */
250 }
251 last_activity_time = sys_timepoint_calc(K_MSEC(LED_TICK_MS * LED_TICKS_ACTIVITY));
252 status_led_can_activity();
253 break;
254
255 case GS_USB_EVENT_CHANNEL_STARTED:
256 LOG_DBG("Channel %u started", ch);
257 last_stopped_time = sys_timepoint_calc(K_NO_WAIT); /* Reset STOPPED filter */
258 /* USB is ready, CAN channel started */
259 status_led_usb_set(LED_USB_READY);
260 status_led_can_set(CAN_LED_ACTIVE);
261 break;
262
263 case GS_USB_EVENT_CHANNEL_STOPPED:
264 /* Filter spurious STOP events from Linux 5/6.1 */
265 if (!sys_timepoint_expired(last_stopped_time)) {
266 LOG_WRN("Channel %u: Ignoring spurious STOPPED event (possible Linux "
267 "kernel bug)",
268 ch);
269 return 0; /* Ignore spurious events */
270 }
271 last_stopped_time = sys_timepoint_calc(K_MSEC(MIN_STOP_INTERVAL_MS));
272 LOG_DBG("Channel %u stopped", ch);
273 /* CAN stopped, USB still ready */
274 status_led_usb_set(LED_USB_READY);
275 status_led_can_set(CAN_LED_OFF);
276 break;
277
278 default:
279 /* Ignore other events */
280 break;
281 }
282
283 return 0;
284}
